ls *mol2
# MOL000006.mol2 MOL000098.mol2

#单个文件格式转换
obabel -imol2 MOL000006.mol2 -opdbqt -O MOL000006.pdbqt

#批量转换
ls *mol2 | while read id; do obabel -imol2 $id -opdbqt -O ${id%.*}.pdbqt; done
