class Graphsage(nn.Module):
	def __init__(self,
				 in_feats,   #输入的样本特征维度(原始特征维度)
				 n_hidden,   #隐藏层样本特征维度
				 n_classes,  #样本的类别数(多分类预测任务)
				 n_layers,   #模型包含几层Graphsage
				 activation, 
				 dropout,    
				 aggregator_type):
		super(Graphsage, self).__init__()
		self.layers = nn.ModuleList()
		self.dropout = nn.Dropout(dropout)
		self.activation = activation
		# 模型第一层：由输入特征到隐藏层特征
		self.layers.append(SAGEConv(in_feats, n_hidden, aggregator_type))
		# 中间隐藏层的变换(如果模型只包含一层，则忽略这一步)
		for i in range(n_layers - 1):
			self.layers.append(SAGEConv(n_hidden, n_hidden, aggregator_type))
		# 模型最后一层：由隐藏层到预测层(不记入总层数)
		self.layers.append(SAGEConv(n_hidden, n_classes, aggregator_type))

	def forward(self, graph, input):
		# 首先对输入特征进行Dropout
		h = self.dropout(input)
		for l, layer in enumerate(self.layers):
			# 进行每一层的GNN节点embedding更新
			h = layer(graph, h)
			# 只要不是最后一层，就添加激活层与Dropout层
			if l != len(self.layers) - 1:
				h = self.activation(h)
				h = self.dropout(h)
		return h

def evaluate(model, graph, features, labels, nid):
	model.eval()
	with torch.no_grad():
		logits = model(graph, features)
		logits = logits[nid]
		labels = labels[nid]
		indices = logits.argmax(1)
		correct = torch.sum(indices == labels)
		return correct.item()*1.0 / len(labels)

data = dgl.data.CoraGraphDataset()
g = data[0]
features = g.ndata['feat']
labels = g.ndata['label']

train_nid = g.ndata['train_mask'].nonzero().squeeze()
val_nid = g.ndata['val_mask'].nonzero().squeeze()

## 模型框架
model = Graphsage(in_feats = g.ndata['feat'].shape[1],
                  n_hidden = 16,
                  n_classes = g.ndata["label"].unique().shape[0],
                  n_layers = 1,
                  activation = F.relu,
                  dropout = 0.5,
                  aggregator_type = "gcn")

## 优化算法
optimizer = torch.optim.Adam(model.parameters(),
							 lr = 0.01,
							 weight_decay = 5e-4)

## 100轮训练
accs = []
for epoch in range(100):
	model.train()
	logits = model(g, features)
	loss = F.cross_entropy(logits[train_nid], labels[train_nid])

	optimizer.zero_grad()
	loss.backward()
	optimizer.step()

	acc = evaluate(model, g, features, labels, val_nid)
	accs.append(acc)
	print("Epoch {:05d} | Loss {:.4f} | Accuracy {:.4f}".format(
										epoch, loss.item(), acc))

import pandas as pd
pd.Series(accs).plot.line()

class SAGE(nn.Module):
	def __init__(self, 
				 in_size,  #输入层维度
				 hid_size, #隐藏层维度
				 out_size):#输出层维度
		super().__init__()
		self.layers = nn.ModuleList()
		self.layers.append(dglnn.SAGEConv(in_size, hid_size, 'mean'))
		self.layers.append(dglnn.SAGEConv(hid_size, hid_size, 'mean'))
		self.layers.append(dglnn.SAGEConv(hid_size, out_size, 'mean'))
		self.dropout = nn.Dropout(0.5)
		self.hid_size = hid_size
		self.out_size = out_size

	def forward(self, blocks, x):
		h = x
		for l, (layer, block) in enumerate(zip(self.layers, blocks)):
			h = layer(block, h)
			# 只要不是最后一层，就添加激活层与Dropout层
			if l != len(self.layers) - 1:
				h = F.relu(h)
				h = self.dropout(h)
		return h

# 每一轮epoch的总体Acc
def evaluate(model, graph, dataloader):
    model.eval()
    ys = []
    y_hats = []
    for it, (input_nodes, output_nodes, blocks) in enumerate(dataloader):
        with torch.no_grad():
            x = blocks[0].srcdata['feat']
            ys.append(blocks[-1].dstdata['label']) #添加每一个block的真实标签
            y_hats.append(model(blocks, x))        #添加每一个block的预测标签
    return MF.accuracy(torch.cat(y_hats), torch.cat(ys))

g.ndata.keys()
g.ndata["feat"].shape
# torch.Size([2708, 1433])

in_size = g.ndata['feat'].shape[1]                 # 1433

model = SAGE(in_size, 256, out_size)
## 优化算法
opt = torch.optim.Adam(model.parameters(),

## 每层采样多少个邻居
sampler = NeighborSampler([10, 10, 10],  # fanout for [layer-0, layer-1, layer-2]
                          prefetch_node_feats=['feat'],
                          prefetch_labels=['label'])
## batchsize大小
train_idx = [i for i, val in enumerate(g.ndata["train_mask"]) if val]

val_idx = [i for i, val in enumerate(g.ndata["val_mask"]) if val]
# 500 
train_dataloader = DataLoader(g, train_idx, sampler, 
                              batch_size=10, shuffle=True,
                              drop_last=False, num_workers=0)
val_dataloader = DataLoader(g, val_idx, sampler, 
                            batch_size=10, shuffle=True,
                            drop_last=False, num_workers=0)

accs = []
for epoch in range(100):
    model.train()
    total_loss = 0
    for it, (input_nodes, output_nodes, blocks) in enumerate(train_dataloader):
        x = blocks[0].srcdata['feat']    #第一层
        y = blocks[-1].dstdata['label']  #最后一层
        y_hat = model(blocks, x)
        loss = F.cross_entropy(y_hat, y)
